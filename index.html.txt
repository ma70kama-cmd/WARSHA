        <input type="tel" id="userPhone" placeholder="رقم الهاتف" required>
        <button type="submit" id="submitBtn">إرسال البيانات</button>
    </form>
    <p id="msg" class="success-msg">تم التسجيل بنجاح!</p>

<script>
    const scriptUrl = 'https://script.google.com/macros/s/AKfycbyzojnq7Q6pBg8wtpxYMWBZF6IOBAAxShD6LT_rb-9eAqS0sKQ01tm33VBY8VPU07HW/exec';

    // إرسال البيانات للجوجل شيت
    document.getElementById('regForm').addEventListener('submit', function(e) {
        e.preventDefault();
        const btn = document.getElementById('submitBtn');
        btn.innerText = "جاري الإرسال...";
        btn.disabled = true;

        const payload = {
            name: document.getElementById('userName').value,
            phone: document.getElementById('userPhone').value
        };

        fetch(scriptUrl, {
            method: 'POST',
            mode: 'no-cors',
            body: JSON.stringify(payload)
        }).then(() => {
            document.getElementById('msg').style.display = 'block';
            btn.innerText = "تم الإرسال";
            setTimeout(() => location.reload(), 2000); // تحديث الصفحة لرؤية الاسم الجديد
        });
    });

    // جلب البيانات لعرضها في الجدول
    async function loadData() {
        try {
            const res = await fetch(scriptUrl);
            const data = await res.json();
            const tbody = document.getElementById('tableBody');
            data.slice(-5).reverse().forEach(item => { // عرض آخر 5 عملاء فقط
                tbody.innerHTML += `<tr><td>${item.name}</td><td>${item.phone}</td></tr>`;
            });
        } catch(e) {}
    }
    loadData();
</script>
